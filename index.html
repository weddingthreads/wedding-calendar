// ============================================ // Google Apps Script - 결혼일 캘린더 API // ============================================ // 사용법: // 1. Google Sheets에서 "확장 프로그램 > Apps Script" 클릭 // 2. 이 코드를 전체 복사하여 붙여넣기 // 3. "배포 > 배포 관리 > 연필 아이콘 > 새 버전 > 배포" // ============================================ const SHEET_NAME = 'entries'; function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const params = e.parameter; const action = params.action; let result; try { switch (action) { case 'read': result = readEntries(); break; case 'add': result = addEntry(params); break; case 'update': result = updateEntry(params); break; case 'delete': result = deleteEntry(params); break; case 'attend': result = attendEntry(params); break; case 'unattend': result = unattendEntry(params); break; default: result = { success: false, error: 'Unknown action' }; } } catch (err) { result = { success: false, error: err.message }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(['id', 'name', 'date', 'time', 'spouse', 'contact', 'memo', 'attendees', 'createdAt']); sheet.getRange(1, 1, 1, 9).setFontWeight('bold'); } // 기존 시트에 attendees 컬럼이 없으면 추가 const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; if (headers.indexOf('attendees') === -1) { const nextCol = sheet.getLastColumn() + 1; sheet.getRange(1, nextCol).setValue('attendees').setFontWeight('bold'); } return sheet; } function getAttendeesCol(sheet) { const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; return headers.indexOf('attendees') + 1; // 1-based } function readEntries() { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return { success: true, entries: [] }; const headers = data[0]; const entries = []; for (let i = 1; i < data.length; i++) { const row = data[i]; if (!row[0]) continue; const entry = {}; headers.forEach((h, idx) => { entry[h] = row[idx]; }); if (entry.date instanceof Date) { entry.date = Utilities.formatDate(entry.date, Session.getScriptTimeZone(), 'yyyy-MM-dd'); } if (entry.createdAt instanceof Date) { entry.createdAt = entry.createdAt.toISOString(); } if (entry.time instanceof Date) { entry.time = Utilities.formatDate(entry.time, Session.getScriptTimeZone(), 'HH:mm'); } else if (typeof entry.time === 'string' && entry.time.includes('1899')) { entry.time = ''; } // attendees는 문자열 그대로 유지 (쉼표 구분) if (!entry.attendees) entry.attendees = ''; entries.push(entry); } return { success: true, entries }; } function addEntry(params) { const sheet = getSheet(); const id = new Date().getTime().toString(36) + Math.random().toString(36).slice(2, 6); const now = new Date().toISOString(); const attCol = getAttendeesCol(sheet); // 기본 행 추가 const row = [ id, params.name || '', params.date || '', params.time || '', params.spouse || '', params.contact || '', params.memo || '' ]; // attendees 컬럼 위치에 맞춰 빈값 채우기 while (row.length < attCol - 1) row.push(''); row.push(''); // attendees while (row.length < attCol) row.push(''); // createdAt const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const createdAtCol = headers.indexOf('createdAt'); while (row.length <= createdAtCol) row.push(''); row[createdAtCol] = now; sheet.appendRow(row); return { success: true, entry: { id, name: params.name, date: params.date, time: params.time || '', spouse: params.spouse, contact: params.contact, memo: params.memo, attendees: '', createdAt: now } }; } function updateEntry(params) { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); const headers = data[0]; const id = params.id; for (let i = 1; i < data.length; i++) { if (data[i][0] === id) { const colMap = {}; headers.forEach((h, idx) => { colMap[h] = idx + 1; }); if (params.name && colMap.name) sheet.getRange(i + 1, colMap.name).setValue(params.name); if (params.date && colMap.date) sheet.getRange(i + 1, colMap.date).setValue(params.date); if (params.time !== undefined && colMap.time) sheet.getRange(i + 1, colMap.time).setValue(params.time); if (params.spouse !== undefined && colMap.spouse) sheet.getRange(i + 1, colMap.spouse).setValue(params.spouse); if (params.contact !== undefined && colMap.contact) sheet.getRange(i + 1, colMap.contact).setValue(params.contact); if (params.memo !== undefined && colMap.memo) sheet.getRange(i + 1, colMap.memo).setValue(params.memo); return { success: true }; } } return { success: false, error: 'Entry not found' }; } function deleteEntry(params) { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); const id = params.id; for (let i = 1; i < data.length; i++) { if (data[i][0] === id) { sheet.deleteRow(i + 1); return { success: true }; } } return { success: false, error: 'Entry not found' }; } // 참여자 추가 function attendEntry(params) { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); const id = params.id; const nickname = params.nickname; if (!nickname) return { success: false, error: 'Nickname required' }; const attCol = getAttendeesCol(sheet); if (attCol === 0) return { success: false, error: 'No attendees column' }; for (let i = 1; i < data.length; i++) { if (data[i][0] === id) { const current = String(data[i][attCol - 1] || ''); const list = current ? current.split(',').map(s => s.trim()).filter(Boolean) : []; if (list.includes(nickname)) { return { success: false, error: 'Already attending' }; } list.push(nickname); sheet.getRange(i + 1, attCol).setValue(list.join(',')); return { success: true, attendees: list.join(',') }; } } return { success: false, error: 'Entry not found' }; } // 참여자 제거 function unattendEntry(params) { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); const id = params.id; const nickname = params.nickname; const attCol = getAttendeesCol(sheet); if (attCol === 0) return { success: false, error: 'No attendees column' }; for (let i = 1; i < data.length; i++) { if (data[i][0] === id) { const current = String(data[i][attCol - 1] || ''); const list = current ? current.split(',').map(s => s.trim()).filter(Boolean) : []; const filtered = list.filter(a => a !== nickname); sheet.getRange(i + 1, attCol).setValue(filtered.join(',')); return { success: true, attendees: filtered.join(',') }; } } return { success: false, error: 'Entry not found' }; }
